function updateSuccess(msg) {
    $(function() {
        if (msg) {
            toastr["success"](msg);
        } else {
            toastr["success"]('global.success');
        }
    })
}
function updateError(msg) {
    $(function() {
        if (msg) {
            toastr["error"](msg);
        } else {
            toastr["error"]('global.error');
        }
    })
}
function warningMessage(msg) {
    $(function() {
        toastr["warning"](msg);
    })
}
function processReturnMessage(returnCode, msg, callback, extraValue) {
    try {
        if (returnCode == 0 && msg != '') {
            if (msg) {
                updateSuccess(msg);
            } else {
                updateSuccess();
            }
        } else if (returnCode == 1 && msg != '') {
            if (msg) {
                updateError(msg);
            } else {
                updateError();
            }
        } else if (returnCode == 2 && msg != '') {
            if (msg) {
                warningMessage(msg);
            }
        }
        if (callback != null && callback != '') {
            setTimeout(function() {
                eval(callback + '(returnCode,extraValue,msg)');
            }, 50)
        }
    } catch (ex) {
        alert(ex.message);
    }
}

}

    function processSearchGroup() {
        if ($('#groupFormSearch').validationEngine('validate')) {
            var areaId = "searchResult";
            var actionUrl = "/VRS/groups/search?";
            $("#groupFormSearch").val($("#groupFormSearch").val().trim());
            var category = $('#category').val();
            var formData = $('#groupFormSearch').serializeArray();
            if (Array.isArray(formData)) {
                formData.push({"name": "category", "value": category});
            }
            dttUpdateAjaxCheckToken("GET", areaId, actionUrl, formData, null, 'Token hết hạn');
        }
    }

        processSearchGroup();
    });

    function openGroupModal() {
        var areaId = "groupModal";
        var actionUrl = "/VRS/groups/add";
        var formData = "";
        dttUpdateAjaxCheckToken("GET", areaId, actionUrl, formData, null, 'Token hết hạn');
    }

        var areaId = "groupEditModal";
        var actionUrl = "/VRS/groups/" + id + "/edit";

        var message = 'Bạn có muốn xóa bản ghi hiện tại?';
        if (id && dttConfirmDelete(message)) {
            var areaId = "saveResultDisplayTag";
            var actionUrl = "/VRS/groups/delete/" + id;
            dttDeferredAjaxCheckToken("POST", areaId, actionUrl, "", "", 'Token hết hạn').done(function () {
                callbackAfterSaveGroup();
            });
        }
    }

    function processSaveGroup() {
        var message = 'Bạn có muốn lưu lại thông tin?';
        var lst = $('#categoryId').val();
        var valid = $('#catGroupForm').validationEngine('validate')
        if (lst == null || lst == undefined || lst.length < 2) {
            document.getElementById('checkCategory').style.display = 'block';
        } else {
            document.getElementById('checkCategory').style.display = 'none';
            if (valid && dttConfirmSave(message)) {
                var areaId = "saveResultDisplayTag";
                var actionUrl = "/VRS/groups/add";

                var formData = $('#catGroupForm').serializeArray();
                if (Array.isArray(formData)) {
                    formData.push({"name": "categoryId", "value": lst});
                }
                dttUpdateAjaxCheckToken("POST", areaId, actionUrl, formData, null, 'Token hết hạn');
            }
        }
    }

    function processEditGroup(id) {
        var message = 'Bạn có muốn lưu lại thông tin?';

        if (lst == null || lst == undefined || lst.length < 2) {
            document.getElementById('checkCategoryEdit').style.display = 'block';
        } else {
            document.getElementById('checkCategoryEdit').style.display = 'none';

                var areaId = "saveResultDisplayTag";
                var actionUrl = "/VRS/groups/" + id + "/edit";
                var formData = $('#catGroupEditForm').serializeArray();

                    formData.push({"name": "categoryId", "value": lst});
                }
                dttUpdateAjaxCheckToken("POST", areaId, actionUrl, formData, null, 'Token hết hạn');
            }
        }
    }
